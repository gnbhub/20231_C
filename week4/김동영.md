int main(void)
{
  int number[5];
  int min;
  for(int i=0;i<5;i++)
    {
      scanf("%d",&number[i]);
    }
   min =number[0];
  for(int i=0;i<5;i++)
    {
      if(min<number[i])
        min=number[0];
      else
        min=number[i];
    }
  printf("최솟값은 %d",min);
return 0;
}
